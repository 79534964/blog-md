## defineProperty
** 特点 **

** 1.不能新增不存在的属性，因为不存在的属性不能被劫持到（get，set） **
** 2.深度响应式 因为每次赋值都会递归对这个对象进行劫持（get，set） **

```javascript
let myVue = new MyVue({
	el：'#app',
	data: {a ：1}
});

// 一个简单的数据劫持
function MyVue(options = {}) {
	this.$options = options;
	var data = this._data = this.$options.data;
	observe(data);
	// 此处代理_data
	for(let key in data) {
		Object.defineProperty(this, key, {
			// 可遍历的
			enumerable: true,
			get() {
				// 此处代理_data到this
				return this._data[key];
			},
			set(newVal) {
				// 此处代理_data到this
				this._data[key] = newVal;
			}
		});
	}
	new Compile(options.el, this);
}
```

## observe

```javascript

// 此处用到了适配器模式
function observe(data) {
	if(typeof data !== 'object') return;
	return Observe(data);
}

// 观察对象增加Object.defineProperty
function Observe(data) {
	let dep = new Dep();
	// 遍历对象
	for(let key in data) {
		// 用来保存每个被劫持的值
		let val = data[key];
		observe(val);
		Object.defineProperty(data, key, {
			// 可遍历的
			enumerable: true,
			get() {
				Dep.target && dep.addSub(Dep.target); // [watcher]
				return val;
			},
			set(newVal) {
				// 判断和之前的值是否一样
				if(newVal === val) {
					return;
				}
				val = newVal;
				observe(newVal);
				// 让所有的watch的update方法执行
				dep.notify();
			}
		});
	}
}
```

## Compile
** 特点 **

** 1.通过createDocumentFragment创建一个虚拟dom将dom节点放到内存中 **

```javascript
function Compile(el, vm) {
	// el表示替换的dom范围
	// vm表示vue实例
	vm.$el = document.querySelector(el);
	let fragment = document.createDocumentFragment();
	// 将dom放入内存中
	while(child = vm.$el.firstChild) {
		fragment.append(child);
	}
	replace(fragment);
	function replace(fragment) {
		Array.from(fragment.childNodes).forEach((node) => {
			let text = node.textContext;
			let reg = /\{\{(.*)\}\}/;
			// 判断是否是文本节点
			if(node.nodeType === 3 && reg.test(text)) {
				// RegExp.$1 a.a
				//  函数需要接收一个新值
				new Watcher(vm, RegExp.$1, function(newVal) {
					node.textContent = text.replace(reg, newVal);
				});
			}
			if(node.childNodes) {
				replace(node);
			}
		});
	}
}
```

## Dep

```javascript
function Dep() {
	this.subs = [];
}
Dep.prototype.addSub = function (sub) {
	this.subs.push(sub);
};
Dep.prototype.notify = function () {
	this.subs.forEach(sub => sub.update());
};
```

## Watcher

```javascript
function Watcher(vm, exp, fn) {
	this.fn = fn;
	this.vm = vm;
	this.exp = exp;
	Dep.target = this;
}
Watcher.prototype.update = function () {
	let val = this.vm;
	let arr =this. exp.split('.');
	arr.forEach((k)=> { // this.a.a
		val = val[k];
	});
	this.fn(val);
}
```
